use super::{expression::Expression, variable::VariableDecl};
use crate::common::display;
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(
    fmt = "predicate {}({}){}\n",
    "name",
    "display::cjoin(parameters)",
    "display::option!(body, \" {{\n  {}\n}}\", \";\")"
)]
pub struct PredicateDecl {
    pub name: String,
    pub parameters: Vec<VariableDecl>,
    pub body: Option<Expression>,
}
impl PartialEq for PredicateDecl {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.name == other.name && self.parameters == other.parameters && self.body == other.body
    }
}
impl core::hash::Hash for PredicateDecl {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.name.hash(state);
        self.parameters.hash(state);
        self.body.hash(state);
    }
}
impl PredicateDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new<G0: Into<String>, G1: Into<Vec<VariableDecl>>, G2: Into<Option<Expression>>>(
        name: G0,
        parameters: G1,
        body: G2,
    ) -> Self {
        Self {
            name: name.into(),
            parameters: parameters.into(),
            body: body.into(),
        }
    }
}
impl PredicateDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new_with_pos<
        G0: Into<String>,
        G1: Into<Vec<VariableDecl>>,
        G2: Into<Option<Expression>>,
    >(
        name: G0,
        parameters: G1,
        body: G2,
    ) -> Self {
        Self {
            name: name.into(),
            parameters: parameters.into(),
            body: body.into(),
        }
    }
}
