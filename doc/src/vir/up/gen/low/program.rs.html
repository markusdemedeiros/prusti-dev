use super::{
    ast::{function::FunctionDecl, predicate::PredicateDecl},
    cfg::{method::MethodDecl, procedure::ProcedureDecl},
    domain::DomainDecl,
};
use crate::common::{check_mode::CheckMode, display};
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(
    fmt = "program_name: {}\n{}\n{}\n{}\n{}\n{}\n",
    name,
    "display::foreach!(\"{}\n\", domains)",
    "display::foreach!(\"{}\n\", predicates)",
    "display::foreach!(\"{}\n\", functions)",
    "display::foreach!(\"{}\n\", methods)",
    "display::foreach!(\"{}\n\", procedures)"
)]
pub struct Program {
    pub name: String,
    pub check_mode: CheckMode,
    pub domains: Vec<DomainDecl>,
    pub predicates: Vec<PredicateDecl>,
    pub functions: Vec<FunctionDecl>,
    pub methods: Vec<MethodDecl>,
    pub procedures: Vec<ProcedureDecl>,
}
impl PartialEq for Program {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.name == other.name
            && self.check_mode == other.check_mode
            && self.domains == other.domains
            && self.predicates == other.predicates
            && self.functions == other.functions
            && self.methods == other.methods
            && self.procedures == other.procedures
    }
}
impl core::hash::Hash for Program {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.name.hash(state);
        self.check_mode.hash(state);
        self.domains.hash(state);
        self.predicates.hash(state);
        self.functions.hash(state);
        self.methods.hash(state);
        self.procedures.hash(state);
    }
}
impl Program {
    #[allow(clippy::too_many_arguments)]
    pub fn new<
        G0: Into<String>,
        G1: Into<CheckMode>,
        G2: Into<Vec<DomainDecl>>,
        G3: Into<Vec<PredicateDecl>>,
        G4: Into<Vec<FunctionDecl>>,
        G5: Into<Vec<MethodDecl>>,
        G6: Into<Vec<ProcedureDecl>>,
    >(
        name: G0,
        check_mode: G1,
        domains: G2,
        predicates: G3,
        functions: G4,
        methods: G5,
        procedures: G6,
    ) -> Self {
        Self {
            name: name.into(),
            check_mode: check_mode.into(),
            domains: domains.into(),
            predicates: predicates.into(),
            functions: functions.into(),
            methods: methods.into(),
            procedures: procedures.into(),
        }
    }
}
impl Program {
    #[allow(clippy::too_many_arguments)]
    pub fn new_with_pos<
        G0: Into<String>,
        G1: Into<CheckMode>,
        G2: Into<Vec<DomainDecl>>,
        G3: Into<Vec<PredicateDecl>>,
        G4: Into<Vec<FunctionDecl>>,
        G5: Into<Vec<MethodDecl>>,
        G6: Into<Vec<ProcedureDecl>>,
    >(
        name: G0,
        check_mode: G1,
        domains: G2,
        predicates: G3,
        functions: G4,
        methods: G5,
        procedures: G6,
    ) -> Self {
        Self {
            name: name.into(),
            check_mode: check_mode.into(),
            domains: domains.into(),
            predicates: predicates.into(),
            functions: functions.into(),
            methods: methods.into(),
            procedures: procedures.into(),
        }
    }
}
