use crate::{
    common::display,
    low::ast::{expression::Expression, ty::Type, variable::VariableDecl},
};
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(
    fmt = "domain {} {{\n{}{} }}\n",
    name,
    "display::foreach!(\"  {}\n\", functions)",
    "display::foreach!(\"  {}\n\", axioms)"
)]
pub struct DomainDecl {
    pub name: String,
    pub functions: Vec<DomainFunctionDecl>,
    pub axioms: Vec<DomainAxiomDecl>,
}
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(
    fmt = "function {}({}): {}",
    name,
    "display::cjoin(parameters)",
    return_type
)]
pub struct DomainFunctionDecl {
    pub name: String,
    pub is_unique: bool,
    pub parameters: Vec<VariableDecl>,
    pub return_type: Type,
}
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(fmt = "axiom {} {{\n  {}\n}}", name, body)]
pub struct DomainAxiomDecl {
    pub name: String,
    pub body: Expression,
}
impl PartialEq for DomainDecl {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.name == other.name && self.functions == other.functions && self.axioms == other.axioms
    }
}
impl core::hash::Hash for DomainDecl {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.name.hash(state);
        self.functions.hash(state);
        self.axioms.hash(state);
    }
}
impl DomainDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new<
        G0: Into<String>,
        G1: Into<Vec<DomainFunctionDecl>>,
        G2: Into<Vec<DomainAxiomDecl>>,
    >(
        name: G0,
        functions: G1,
        axioms: G2,
    ) -> Self {
        Self {
            name: name.into(),
            functions: functions.into(),
            axioms: axioms.into(),
        }
    }
}
impl DomainDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new_with_pos<
        G0: Into<String>,
        G1: Into<Vec<DomainFunctionDecl>>,
        G2: Into<Vec<DomainAxiomDecl>>,
    >(
        name: G0,
        functions: G1,
        axioms: G2,
    ) -> Self {
        Self {
            name: name.into(),
            functions: functions.into(),
            axioms: axioms.into(),
        }
    }
}
impl PartialEq for DomainFunctionDecl {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.name == other.name
            && self.is_unique == other.is_unique
            && self.parameters == other.parameters
            && self.return_type == other.return_type
    }
}
impl core::hash::Hash for DomainFunctionDecl {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.name.hash(state);
        self.is_unique.hash(state);
        self.parameters.hash(state);
        self.return_type.hash(state);
    }
}
impl DomainFunctionDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new<G0: Into<String>, G1: Into<bool>, G2: Into<Vec<VariableDecl>>, G3: Into<Type>>(
        name: G0,
        is_unique: G1,
        parameters: G2,
        return_type: G3,
    ) -> Self {
        Self {
            name: name.into(),
            is_unique: is_unique.into(),
            parameters: parameters.into(),
            return_type: return_type.into(),
        }
    }
}
impl DomainFunctionDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new_with_pos<
        G0: Into<String>,
        G1: Into<bool>,
        G2: Into<Vec<VariableDecl>>,
        G3: Into<Type>,
    >(
        name: G0,
        is_unique: G1,
        parameters: G2,
        return_type: G3,
    ) -> Self {
        Self {
            name: name.into(),
            is_unique: is_unique.into(),
            parameters: parameters.into(),
            return_type: return_type.into(),
        }
    }
}
impl PartialEq for DomainAxiomDecl {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.name == other.name && self.body == other.body
    }
}
impl core::hash::Hash for DomainAxiomDecl {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.name.hash(state);
        self.body.hash(state);
    }
}
impl DomainAxiomDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new<G0: Into<String>, G1: Into<Expression>>(name: G0, body: G1) -> Self {
        Self {
            name: name.into(),
            body: body.into(),
        }
    }
}
impl DomainAxiomDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new_with_pos<G0: Into<String>, G1: Into<Expression>>(name: G0, body: G1) -> Self {
        Self {
            name: name.into(),
            body: body.into(),
        }
    }
}
