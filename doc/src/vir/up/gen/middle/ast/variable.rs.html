use super::ty::Type;
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(fmt = "{}: {}", name, ty)]
pub struct VariableDecl {
    pub name: String,
    pub ty: Type,
}
impl PartialEq for VariableDecl {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.name == other.name && self.ty == other.ty
    }
}
impl core::hash::Hash for VariableDecl {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.name.hash(state);
        self.ty.hash(state);
    }
}
impl Ord for VariableDecl {
    #[allow(unused_variables)]
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        self.name.cmp(&other.name).then(self.ty.cmp(&other.ty))
    }
}
impl PartialOrd for VariableDecl {
    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
        Some(self.cmp(other))
    }
}
impl VariableDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new<G0: Into<String>, G1: Into<Type>>(name: G0, ty: G1) -> Self {
        Self {
            name: name.into(),
            ty: ty.into(),
        }
    }
}
impl VariableDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new_with_pos<G0: Into<String>, G1: Into<Type>>(name: G0, ty: G1) -> Self {
        Self {
            name: name.into(),
            ty: ty.into(),
        }
    }
}
